99int cp_cls_ver(const g1_t a, const g1_t b, const g1_t c, const uint8_t *msg,
100 size_t len, const g2_t x, const g2_t y) {
101 g1_t p[2];
102 g2_t r[2];
103 gt_t e;
104 bn_t m, n;
105 int result = 1;
106
107 g1_null(p[0]);
108 g1_null(p[1]);
109 g2_null(r[0]);
110 g2_null(r[1]);
111 gt_null(e);
112 bn_null(m);
113 bn_null(n);
114
115 RLC_TRY {
116 g1_new(p[0]);
117 g1_new(p[1]);
118 g2_new(r[0]);
119 g2_new(r[1]);
120 gt_new(e);
121 bn_new(m);
122 bn_new(n);
123
124 if (g1_is_infty(a) || g1_is_infty(b) || g1_is_infty(c)) {
125 result = 0;
126 }
127
128 g1_copy(p[0], a);
129 g1_copy(p[1], b);
130 g2_copy(r[0], y);
131 g2_get_gen(r[1]);
132 g2_neg(r[1], r[1]);
133
134 pc_map_sim(e, p, r, 2);
135 if (!gt_is_unity(e)) {
136 result = 0;
137 }
138
139 pc_get_ord(n);
140 bn_read_bin(m, msg, len);
141 bn_mod(m, m, n);
142 g1_mul(p[0], b, m);
143 g1_add(p[0], p[0], a);
144 g1_norm(p[0], p[0]);
145 g1_copy(p[1], c);
146 g2_copy(r[0], x);
147
148 pc_map_sim(e, p, r, 2);
149 if (!gt_is_unity(e)) {
150 result = 0;
151 }
152 }
153 RLC_CATCH_ANY {
154 RLC_THROW(ERR_CAUGHT);
155 }
156 RLC_FINALLY {
157 g1_free(p[0]);
158 g1_free(p[1]);
159 g2_free(r[0]);
160 g2_free(r[1]);
161 gt_free(e);
162 bn_free(m);
163 bn_free(n);
164 }
165 return result;
166}

194int cp_cli_sig(g1_t a, g1_t A, g1_t b, g1_t B, g1_t c, const uint8_t *msg,
195 size_t len, const bn_t r, const bn_t t, const bn_t u, const bn_t v) {
196 bn_t m, n;
197 int result = RLC_OK;
198
199 bn_null(m);
200 bn_null(n);
201
202 RLC_TRY {
203 bn_new(m);
204 bn_new(n);
205
206 /* Choose random a in G1. */
207 g1_rand(a);
208 /* Compute A = a^z, B = A^y. */
209 g1_mul(A, a, v);
210 g1_mul(b, a, u);
211 g1_mul(B, A, u);
212 /* Compute c = A^(xyr) = B^{xr}. */
213 pc_get_ord(n);
214 bn_mul(m, t, r);
215 bn_mod(m, m, n);
216 g1_mul(b, B, m);
217 /* Compute c = a^{x+xym}A^(xyr). */
218 bn_read_bin(m, msg, len);
219 bn_mod(m, m, n);
220 bn_mul(m, m, t);
221 bn_mod(m, m, n);
222 bn_mul(m, m, u);
223 bn_mod(m, m, n);
224 bn_add(m, m, t);
225 bn_mod(m, m, n);
226 g1_mul(c, a, m);
227 g1_add(c, c, b);
228 g1_norm(c, c);
229 /* Compute b = a^y. */
230 g1_mul(b, a, u);
231 }
232 RLC_CATCH_ANY {
233 result = RLC_ERR;
234 }
235 RLC_FINALLY {
236 bn_free(m);
237 bn_free(n);
238 }
239 return result;
240}

242int cp_cli_ver(g1_t a, g1_t A, g1_t b, g1_t B, g1_t c, const uint8_t *msg,
243 size_t len, const bn_t r, const g2_t x, const g2_t y, const g2_t z) {
244 g1_t p[2];
245 g2_t q[2];
246 gt_t e;
247 bn_t m, n;
248 int result = 1;
249
250 g1_null(p[0]);
251 g1_null(p[1]);
252 g2_null(q[0]);
253 g2_null(q[1]);
254 gt_null(e);
255 bn_null(m);
256 bn_null(n);
257
258 RLC_TRY {
259 g1_new(p[0]);
260 g1_new(p[1]);
261 g2_new(q[0]);
262 g2_new(q[1]);
263 gt_new(e);
264 bn_new(m);
265 bn_new(n);
266
267 if (g1_is_infty(a) || g1_is_infty(A)) {
268 result = 0;
269 }
270 if (g1_is_infty(b) || g1_is_infty(B) || g1_is_infty(c)) {
271 result = 0;
272 }
273
274 /* Check that e(a, Z) = e(A, g) or that e(a, Z)e(A, -g) = 1. */
275 g1_copy(p[0], a);
276 g1_copy(p[1], A);
277 g2_copy(q[0], z);
278 g2_get_gen(q[1]);
279 g2_neg(q[1], q[1]);
280 pc_map_sim(e, p, q, 2);
281 if (!gt_is_unity(e)) {
282 result = 0;
283 }
284
285 /* Check e(a, Y) = e(b, g) and e(A, Y) = e(B, g) using same trick. */
286 g1_copy(p[1], b);
287 g2_copy(q[0], y);
288 pc_map_sim(e, p, q, 2);
289 if (!gt_is_unity(e)) {
290 result = 0;
291 }
292 g1_copy(p[0], A);
293 g1_copy(p[1], B);
294 pc_map_sim(e, p, q, 2);
295 if (!gt_is_unity(e)) {
296 result = 0;
297 }
298
299 /* Check that e(a, X)e(mb, X)e(rB, X) = e(c, g). */
300 pc_get_ord(n);
301 bn_read_bin(m, msg, len);
302 bn_mod(m, m, n);
303 g1_mul_sim(p[0], b, m, B, r);
304 g1_add(p[0], p[0], a);
305 g1_norm(p[0], p[0]);
306 g1_copy(p[1], c);
307 g2_copy(q[0], x);
308 pc_map_sim(e, p, q, 2);
309 if (!gt_is_unity(e)) {
310 result = 0;
311 }
312 }
313 RLC_CATCH_ANY {
314 RLC_THROW(ERR_CAUGHT);
315 }
316 RLC_FINALLY {
317 g1_free(p[0]);
318 g1_free(p[1]);
319 g2_free(q[1]);
320 g2_free(q[0]);
321 gt_free(e);
322 bn_free(m);
323 bn_free(n);
324 }
325 return result;
326}

356int cp_clb_sig(g1_t a, g1_t A[], g1_t b, g1_t B[], g1_t c, const uint8_t *ms[],
357 const size_t ls[], const bn_t t, const bn_t u, const bn_t v[],
358 size_t l) {
359 bn_t m, n;
360 int i, result = RLC_OK;
361
362 bn_null(m);
363 bn_null(n);
364
365 RLC_TRY {
366 bn_new(m);
367 bn_new(n);
368
369 /* Choose random a in G1. */
370 g1_rand(a);
371 /* Compute A_i = a^z_i, B_i = A_i^y. */
372 for (i = 1; i < l; i++) {
373 g1_mul(A[i - 1], a, v[i - 1]);
374 g1_mul(B[i - 1], A[i - 1], u);
375 }
376 /* Compute c = a^(x+xym_0)\prod A_i^(xym_i) = B_i^(xm_i). */
377 pc_get_ord(n);
378 bn_read_bin(m, ms[0], ls[0]);
379 bn_mod(m, m, n);
380 bn_mul(m, m, t);
381 bn_mod(m, m, n);
382 bn_mul(m, m, u);
383 bn_add(m, m, t);
384 bn_mod(m, m, n);
385 g1_mul(c, a, m);
386 /* This can be made faster with more interleaving. */
387 for (i = 1; i < l; i++) {
388 bn_read_bin(m, ms[i], ls[i]);
389 bn_mod(m, m, n);
390 bn_mul(m, m, t);
391 bn_mod(m, m, n);
392 g1_mul(b, B[i - 1], m);
393 g1_add(c, c, b);
394 }
395 g1_norm(c, c);
396 /* Compute b = a^y. */
397 g1_mul(b, a, u);
398 }
399 RLC_CATCH_ANY {
400 result = RLC_ERR;
401 }
402 RLC_FINALLY {
403 bn_free(m);
404 bn_free(n);
405 }
406 return result;
407}
