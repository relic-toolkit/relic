42void fp24_mul_basic(fp24_t c, const fp24_t a, const fp24_t b) {
43 fp8_t t0, t1, t2, t3, t4, t5;
44
45 fp8_null(t0);
46 fp8_null(t1);
47 fp8_null(t2);
48 fp8_null(t3);
49 fp8_null(t4);
50 fp8_null(t5);
51
52 RLC_TRY {
53 fp8_new(t0);
54 fp8_new(t1);
55 fp8_new(t2);
56 fp8_new(t3);
57 fp8_new(t4);
58 fp8_new(t5);
59
60 /* Karatsuba algorithm. */
61
62 /* t0 = a_0 * b_0. */
63 fp8_mul(t0, a[0], b[0]);
64 /* t1 = a_1 * b_1. */
65 fp8_mul(t1, a[1], b[1]);
66 /* t2 = a_2 * b_2. */
67 fp8_mul(t2, a[2], b[2]);
68
69 fp8_add(t3, a[1], a[2]);
70 fp8_add(t4, b[1], b[2]);
71 fp8_mul(t3, t3, t4);
72 fp8_sub(t3, t3, t1);
73 fp8_sub(t3, t3, t2);
74 fp8_mul_art(t3, t3);
75 fp8_add(t3, t3, t0);
76
77 fp8_add(t4, a[0], a[1]);
78 fp8_add(t5, b[0], b[1]);
79 fp8_mul(t4, t4, t5);
80 fp8_sub(t4, t4, t0);
81 fp8_sub(t4, t4, t1);
82 fp8_mul_art(t5, t2);
83 fp8_add(c[1], t4, t5);
84
85 fp8_add(t4, a[0], a[2]);
86 fp8_add(t5, b[0], b[2]);
87 fp8_mul(c[2], t4, t5);
88 fp8_sub(c[2], c[2], t0);
89 fp8_add(c[2], c[2], t1);
90 fp8_sub(c[2], c[2], t2);
91
92 fp8_copy(c[0], t3);
93 } RLC_CATCH_ANY {
94 RLC_THROW(ERR_CAUGHT);
95 } RLC_FINALLY {
96 fp8_free(t0);
97 fp8_free(t1);
98 fp8_free(t2);
99 fp8_free(t3);
100 fp8_free(t4);
101 fp8_free(t5);
102 }
103}

109void fp24_mul_unr(dv24_t c, const fp24_t a, const fp24_t b) {
110 dv8_t u0, u1, u2, u3, u4;
111 fp8_t t0, t1;
112
113 dv8_null(u0);
114 dv8_null(u1);
115 dv8_null(u2);
116 dv8_null(u3);
117 dv8_null(u4);
118 fp8_null(t0);
119 fp8_null(t1);
120
121 RLC_TRY {
122 dv8_new(u0);
123 dv8_new(u1);
124 dv8_new(u2);
125 dv8_new(u3);
126 dv8_new(u4);
127 fp8_new(t0);
128 fp8_new(t1);
129
130 /* Karatsuba algorithm. */
131
132 /* u0 = a_0 * b_0. */
133 fp8_mul_unr(u0, a[0], b[0]);
134 /* u1 = a_1 * b_1. */
135 fp8_mul_unr(u1, a[1], b[1]);
136 /* u2 = a_2 * b_2. */
137 fp8_mul_unr(u2, a[2], b[2]);
138
139 fp8_add(t0, a[1], a[2]);
140 fp8_add(t1, b[1], b[2]);
141 fp8_mul_unr(u3, t0, t1);
142 for (int i = 0; i < 2; i++) {
143 for (int j = 0; j < 2; j++) {
144 fp2_subc_low(u3[i][j], u3[i][j], u1[i][j]);
145 fp2_subc_low(u3[i][j], u3[i][j], u2[i][j]);
146 }
147 }
148
149 fp2_nord_low(u4[0][0], u3[1][1]);
150 dv_copy(u4[0][1][0], u3[1][0][0], 2 * RLC_FP_DIGS);
151 dv_copy(u4[0][1][1], u3[1][0][1], 2 * RLC_FP_DIGS);
152 for (int j = 0; j < 2; j++) {
153 fp2_addc_low(c[0][0][j], u4[0][j], u0[0][j]);
154 fp2_addc_low(c[0][1][j], u3[0][j], u0[1][j]);
155 }
156
157 fp8_add(t0, a[0], a[1]);
158 fp8_add(t1, b[0], b[1]);
159 fp8_mul_unr(u4, t0, t1);
160 for (int i = 0; i < 2; i++) {
161 for (int j = 0; j < 2; j++) {
162 fp2_subc_low(u4[i][j], u4[i][j], u0[i][j]);
163 fp2_subc_low(u4[i][j], u4[i][j], u1[i][j]);
164 }
165 }
166 fp2_nord_low(u3[0][0], u2[1][1]);
167 dv_copy(u3[0][1][0], u2[1][0][0], 2 * RLC_FP_DIGS);
168 dv_copy(u3[0][1][1], u2[1][0][1], 2 * RLC_FP_DIGS);
169 for (int j = 0; j < 2; j++) {
170 fp2_addc_low(c[1][0][j], u4[0][j], u3[0][j]);
171 fp2_addc_low(c[1][1][j], u4[1][j], u2[0][j]);
172 }
173
174 fp8_add(t0, a[0], a[2]);
175 fp8_add(t1, b[0], b[2]);
176 fp8_mul_unr(u4, t0, t1);
177 for (int i = 0; i < 2; i++) {
178 for (int j = 0; j < 2; j++) {
179 fp2_subc_low(u4[i][j], u4[i][j], u0[i][j]);
180 fp2_addc_low(u4[i][j], u4[i][j], u1[i][j]);
181 fp2_subc_low(c[2][i][j], u4[i][j], u2[i][j]);
182 }
183 }
184 } RLC_CATCH_ANY {
185 RLC_THROW(ERR_CAUGHT);
186 } RLC_FINALLY {
187 dv8_free(u0);
188 dv8_free(u1);
189 dv8_free(u2);
190 dv8_free(u3);
191 dv8_free(u4);
192 fp8_free(t0);
193 fp8_free(t1);
194 }
195}

220void fp24_mul_art(fp24_t c, const fp24_t a) {
221 fp8_t t0;
222
223 fp8_null(t0);
224
225 RLC_TRY {
226 fp8_new(t0);
227
228 /* (a_0 + a_1 * v + a_2 * v^2) * v = a_2 + a_0 * v + a_1 * v^2 */
229 fp8_copy(t0, a[0]);
230 fp8_mul_art(c[0], a[2]);
231 fp8_copy(c[2], a[1]);
232 fp8_copy(c[1], t0);
233 } RLC_CATCH_ANY {
234 RLC_THROW(ERR_CAUGHT);
235 } RLC_FINALLY {
236 fp8_free(t0);
237 }

240void fp24_mul_dxs(fp24_t c, const fp24_t a, const fp24_t b) {
241 fp8_t t0, t1, t2, t3, t4;
242
243 fp8_null(t0);
244 fp8_null(t1);
245 fp8_null(t2);
246 fp8_null(t3);
247 fp8_null(t4);
248
249 RLC_TRY {
250 fp8_new(t0);
251 fp8_new(t1);
252 fp8_new(t2);
253 fp8_new(t3);
254 fp8_new(t4);
255
256 /* Karatsuba algorithm. */
257
258 /* t0 = a_0 * b_0. */
259 fp8_mul(t0, a[0], b[0]);
260 fp8_add(t3, a[1], a[2]);
261 fp8_add(t4, a[0], a[1]);
262
263 if (fp8_is_zero(b[2])) {
264 /* t1 = a_1 * b_1. */
265 fp8_mul(t1, a[1], b[1]);
266 /* b_2 = 0. */
267
268 fp8_mul(t3, t3, b[1]);
269 fp8_sub(t3, t3, t1);
270 fp8_mul_art(t3, t3);
271 fp8_add(t3, t3, t0);
272
273 fp8_add(t2, b[0], b[1]);
274 fp8_mul(t4, t4, t2);
275 fp8_sub(t4, t4, t0);
276 fp8_sub(c[1], t4, t1);
277
278 fp8_add(t4, a[0], a[2]);
279 fp8_mul(c[2], t4, b[0]);
280 fp8_sub(c[2], c[2], t0);
281 fp8_add(c[2], c[2], t1);
282 } else {
283 /* b_1 = 0. */
284 /* t2 = a_2 * b_2. */
285 fp8_mul(t1, a[2], b[2]);
286
287 fp8_mul(t3, t3, b[2]);
288 fp8_sub(t3, t3, t1);
289 fp8_mul_art(t3, t3);
290 fp8_add(t3, t3, t0);
291
292 fp8_mul(t4, t4, b[0]);
293 fp8_sub(t4, t4, t0);
294 fp8_mul_art(t2, t1);
295 fp8_add(c[1], t4, t2);
296
297 fp8_add(t4, a[0], a[2]);
298 fp8_add(t2, b[0], b[2]);
299 fp8_mul(c[2], t4, t2);
300 fp8_sub(c[2], c[2], t0);
301 fp8_sub(c[2], c[2], t1);
302 }
303
304 fp8_copy(c[0], t3);
305 } RLC_CATCH_ANY {
306 RLC_THROW(ERR_CAUGHT);
307 } RLC_FINALLY {
308 fp8_free(t0);
309 fp8_free(t1);
310 fp8_free(t2);
311 fp8_free(t3);
312 fp8_free(t4);
313 }
